JavaScript Code with Async Callback

// Function to get all books using async callback function getAllBooks(callback) { // Simulate API call delay setTimeout(() => { const books = [ { id: 1, title: "The Great Gatsby", author: "F. Scott Fitzgerald", isbn: "9780743273565", price: 12.99, year: 1925 }, { id: 2, title: "To Kill a Mockingbird", author: "Harper Lee", isbn: "9780061120084", price: 11.99, year: 1960 }, { id: 3, title: "1984", author: "George Orwell", isbn: "9780451524935", price: 10.99, year: 1949 }, { id: 4, title: "Pride and Prejudice", author: "Jane Austen", isbn: "9780141439518", price: 9.99, year: 1813 } ]; // Call the callback function with the results callback(null, books); }, 1500); // Simulate 1.5 second API delay } // Usage of the async callback function console.log("Fetching books..."); getAllBooks((error, books) => { if (error) { console.error("Error:", error); return; } console.log("Books received:", books); // Process the books data here });
